#include <stdio.h>
int main()
{

  int T[n];
  printf("enter numbers you want : ");
  for(int i=0;i<n;i++){
    scanf("%d",&T[i]);
  }
max1=T[0];
max2=T[1];

if(max2>max1){
    S=max1;
    max1=max2;
    max2=S;
}
    for(int i=2;i<n;i++){
            if(max1<T[i]){
                max2=max1;
            max1=T[i];
            }

            else if (T[i] > max2 && T[i] != max1) {
            max2 = T[i];
            }
        }

}
